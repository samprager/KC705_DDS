00238 /******************************************************************************
00239  * Move the BdPtr argument ahead an arbitrary number of BDs wrapping around
00240  * to the beginning of the ring if needed.
00241  *
00242  * We know that a wraparound should occur if the new BdPtr is greater than
00243  * the high address in the ring OR if the new BdPtr crosses the 0xFFFFFFFF
00244  * to 0 boundary.
00245  *
00246  * @param RingPtr is the ring BdPtr appears in
00247  * @param BdPtr on input is the starting BD position and on output is the
00248  *        final BD position
00249  * @param NumBd is the number of BD spaces to increment
00250  *
00251  *****************************************************************************/
00252 #define Dma_mRingSeekahead(RingPtr, BdPtr, NumBd)         \
00253   {                   \
00254     u32 Addr = (u32)(BdPtr);            \
00255                       \
00256     Addr += ((RingPtr)->Separation * (NumBd));        \
00257     if ((Addr > (RingPtr)->LastBdAddr) || ((u32)(BdPtr) > Addr))\
00258     {                 \
00259       Addr -= (RingPtr)->Length;          \
00260     }                 \
00261                       \
00262     (BdPtr) = (Dma_Bd*)Addr;            \
00263   }
00264

00266 /******************************************************************************
00267  * Move the BdPtr argument backwards an arbitrary number of BDs wrapping
00268  * around to the end of the ring if needed.
00269  *
00270  * We know that a wraparound should occur if the new BdPtr is less than
00271  * the base address in the ring OR if the new BdPtr crosses the 0xFFFFFFFF
00272  * to 0 boundary.
00273  *
00274  * @param RingPtr is the ring BdPtr appears in
00275  * @param BdPtr on input is the starting BD position and on output is the
00276  *        final BD position
00277  * @param NumBd is the number of BD spaces to increment
00278  *
00279  *****************************************************************************/
00280 #define Dma_mRingSeekback(RingPtr, BdPtr, NumBd)            \
00281   {                                                                     \
00282     u32 Addr = (u32)(BdPtr);              \
00283                         \
00284     Addr -= ((RingPtr)->Separation * (NumBd));          \
00285     if ((Addr < (RingPtr)->FirstBdAddr) || ((u32)(BdPtr) < Addr)) \
00286     {                   \
00287       Addr += (RingPtr)->Length;            \
00288     }                   \
